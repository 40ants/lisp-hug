My mistake, here it is:

---deliver.lisp---
(in-package :cl-user)
(load "/dev/lispworks/.lispworks")
(require :sematix)
(require "overwrite") ;used by editor in insert mode
(in-package :sematix)

(lw:deliver 'make-sematix
   (concatenate 'string "/dev/lispworks/sematix/deliver/sematix-" *sematix-version* ".exe")
   0
   :interface :capi
   :console :input
   :keep-clos :no-empty)

(quit)
---eof deliver.lisp---
