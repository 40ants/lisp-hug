   ()
   (:panes
    (text-input-pane my-text-input-choice)))

(defun my-callback(&rest args)
   args)

(defmethod capi:pane-popup-menu-items ((self my-text-input-choice)

                   :title "Macros"
                   :items (mapcar #'(lambda(x) (make-instance

    (call-next-method)))

(capi:display (make-instance 'my-dialog))
