changing the input-model


Is it possible to change the input-model of a displayed output-pane ?


(defun message (pane x y tpane)
   (setf (capi:title-pane-text tpane) (format nil "(~a,~a)" x y)))

(let* ((mess (make-instance 'capi:title-pane))
        (pane (make-instance 'capi:pinboard-layout
			    :input-model
			    `(((:motion) message ,mess))))
        (layout (make-instance 'capi:column-layout
			      :description (list pane mess))))
   (capi:contain layout)
   (setf (capi:output-pane-input-model pane) nil))
