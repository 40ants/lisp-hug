I found the following at http://paste.lisp.org/display/22379

(defun n-choose-k (list k &optional acc)
  (cond
    ((= k 0) (print acc))
    ((null list))
    (t (n-choose-k (cdr list) k acc)
       (n-choose-k (cdr list) (1- k) (cons (car list) acc)))))
