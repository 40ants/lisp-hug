Text scaling example

Here's an example of what changed from 6.0 to 6.1.

In 6.0 the text is the same size in both windows. In 6.1 it's different.

LW



(defun draw-my-text1 (pane x y w h)
        (gp:with-graphics-state (pane :transform (list 1 0 0 1 0 0))
          (gp:draw-string pane "Test text" 10 50)))


(setq test-pane1
    (capi:contain
         (make-instance 'capi:output-pane
         :display-callback 'draw-my-text1)
         :best-width 300
         :best-height 300))


(defun draw-my-text4 (pane x y w h)
        (gp:with-graphics-state (pane :transform (list 4 0 0 4 0 0))
          (gp:draw-string pane "Test text" 10 50)))


(setq test-pane4
    (capi:contain
         (make-instance 'capi:output-pane
         :display-callback 'draw-my-text4)
         :best-width 300
         :best-height 300))
